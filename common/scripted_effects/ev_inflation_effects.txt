inflation_change_var = {
	
	#基础增长	
		if = {
			limit = {
				AND = {
					check_variable = {
						which = ev_inflation_value
						value > -100
					}
					check_variable = {
						which = ev_inflation_value
						value < 6
					}
				}
			}
			change_variable = {
				which = ev_inflation_value
				value = 0.003
			}
			
		}
		else_if = {
			limit = {
				AND = {
					check_variable = {
						which = ev_inflation_value
						value > 6
					}
					check_variable = {
						which = ev_inflation_value
						value < 20
					}
				}
			}
			multiply_variable = {
				which = ev_inflation_value
				value = 0.9995
			}
			change_variable = {
				which = ev_inflation_value
				value = 0.002
			}
		}
		else_if = {
			limit = {
				AND = {
					check_variable = {
						which = ev_inflation_value
						value >= 20
					}
					check_variable = {
						which = ev_inflation_value
						value < 101
					}
				}
			}
			multiply_variable = {
				which = ev_inflation_value
				value = 0.999
			}
			change_variable = {
				which = ev_inflation_value
				value = 0.004
			}
		}
		else = {
			change_variable = {
				which = ev_inflation_value
				value = 0.001
			}
			set_variable = {
				which = ev_inflation_value_base
				value = 0.001
			}
		}	
		
}
