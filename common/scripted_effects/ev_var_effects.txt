			AND = {
				check_variable = {
					which = ev_maintenance_value
					value > 75
				}
				check_variable = {
					which = ev_maintenance_value
					value <= 100
				}
			}
		}
		change_variable = {
			which = ev_maintenance_value
			value = 0.15
		}
		set_variable = {
			which = ev_maintenance_value_base
			value = 0.15
		}
	}
	else_if = {
		limit = {
			NOT = { has_valid_civic = civic_ev_maintenance_add }
			AND = {
				check_variable = {
					which = ev_maintenance_value
					value <= 75
				}
			}
		}
		change_variable = {
			which = ev_maintenance_value
			value = 0.1
		}
		set_variable = {
			which = ev_maintenance_value_base
			value = 0.1
		}
	}
	else_if = {
		limit = {
			has_valid_civic = civic_ev_maintenance_add
			AND = {
				check_variable = {
					which = ev_maintenance_value
					value <= 75
				}
			}
		}
		change_variable = {
			which = ev_maintenance_value
			value = 0.3
		}
		set_variable = {
			which = ev_maintenance_value_base
			value = 0.3
		}
	}
	else = {
		change_variable = {
			which = ev_maintenance_value
			value = 0.1
		}
		set_variable = {
			which = ev_maintenance_value_base
			value = 0.1
		}
	}
	
	
	if = {
		limit = { 
			has_policy_flag = ev_maintenance_reduce_low
		}
		set_variable = {
			which = ev_maintenance_value_policy
			value = -0.15
		}
		change_variable = {
			which = ev_maintenance_value
			value = -0.15
		}
	}
	else_if = {
		limit = { 
			has_policy_flag = ev_maintenance_reduce_high
		}
		set_variable = {
			which = ev_maintenance_value_policy
			value = -0.3
		}
		change_variable = {
			which = ev_maintenance_value
			value = -0.3
		}
	}
	else = {
		set_variable = {
			which = ev_maintenance_value_policy
			value = 0
		}
	}
	remove_ev_maintenance_buff = yes
	give_ev_maintenance_buff = yes
	
	
}
