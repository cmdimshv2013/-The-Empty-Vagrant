# 设定虫子来的事件
country_event = {
	id = ev_crisis_other.1
	hide_window = yes
	#fire_only_once = yes
	
	mean_time_to_happen = {
		months = 1
	}
	
	trigger = {
		OR = {
			is_country_type = default_ev
			is_country_type = fallen_empire_ev
			is_country_type = awakened_fallen_empire_ev
		}
		has_global_flag = prethoryn_subspace_echoes
		NOT = { has_country_flag = has_trigger_crisis_10 }
	}
	
	immediate = {
		set_country_flag = has_trigger_crisis_10
		begin_event_chain = {
			event_chain = "coming_storm_chain"
			target = ROOT
		}
	}
}

	id = ev_crisis_other.8
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		OR = {
			is_country_type = extradimensional
			is_country_type = extradimensional_2
			is_country_type = extradimensional_3
		}
		NOT = {
			any_country = {
				AND = {
					NOT = { is_same_value = ROOT }
					OR = {
						is_country_type = extradimensional
						is_country_type = extradimensional_2
						is_country_type = extradimensional_3
					}
				}
			}
		}
	}
	
	immediate = {
		observer_event = { id = observer.40 }
		every_country = {
			limit = {
				OR = {
					is_country_type = default_ev

					is_country_type = awakened_fallen_empire_ev
				}
			}
			country_event = { id = crisis.1271 }
		}
	}
}

#击败肃正屑议
planet_event = {
	id = ev_crisis_other.11
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		is_planet_class = pc_ai
		has_planet_flag = machine_lair
		OR = { # cracked or bombarded to death
			from = { is_same_value = root }
			planet_devastation >= 50
		}
	}
	
	immediate = {
		every_country = {
			limit = {
				OR = {
					is_country_type = default_ev
					is_country_type = fallen_empire_ev
					is_country_type = awakened_fallen_empire_ev
				}
			}
			stop_crisis_sound = yes
			from = { save_event_target_as = final_machine_world_destroyer }
			planet_event = { id = crisis.2046 }
		}
	}
}
