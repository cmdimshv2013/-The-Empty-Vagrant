country_event = {
	id = ev_resource.1
	
	hide_window = yes
	
	location = root
	
	#is_triggered_only = yes
	
	mean_time_to_happen = {
		months = 1
	}
	
	trigger = {
		owner = {
			is_country_type = default_ev
		}
	}
	
	immediate = {
		owner = {
			remove_modifier = empty_vagrant_no_minerals 
			remove_modifier = empty_vagrant_no_consumer_goods
			remove_modifier = empty_vagrant_no_alloys
			remove_modifier = empty_vagrant_no_energy
			remove_modifier = empty_vagrant_no_food
			remove_modifier = empty_vagrant_no_advanced_resources
			remove_modifier = empty_vagrant_no_volatile_motes
			remove_modifier = empty_vagrant_no_exotic_gases
			remove_modifier = empty_vagrant_no_rare_crystals
			remove_modifier = empty_vagrant_no_influence
		}
		if = {
			limit = {
				owner = {	
					AND = {
					is_country_type = default_ev
					has_country_resource = { type = influence amount <= 0 }
					has_monthly_income = { resource = influence value < 0 }
					}
				}
			}
			add_modifier = { modifier = empty_vagrant_no_influence days = -1 } 
		}
		if = {
			limit = {
				owner = {	
					AND = {
					is_country_type = default_ev
					has_country_resource = { type = minerals amount < 1 }
					has_monthly_income = { resource = minerals value < 0 }
					}
				}
			}
			add_modifier = { modifier = empty_vagrant_no_minerals days = -1 } 
		}
		if = {
			limit = {
				owner = {	
					AND = {
						is_country_type = default_ev
						NOT = { is_lithoid_empire = yes }
						has_country_resource = { type = food amount < 1 }
						has_monthly_income = { resource = food value < 0 }
					}
				}
			}
			add_modifier = { modifier = empty_vagrant_no_food days = -1 } 
		}
		if = {
			limit = {
				owner = {	
					AND = {
						NOR = {
							has_authority = auth_machine_intelligence
							has_authority = auth_hive_mind
						}
						is_country_type = default_ev
						has_country_resource = { type = consumer_goods amount < 1 }
						has_monthly_income = { resource = consumer_goods value < 0 }
					}
				}
			}
			add_modifier = { modifier = empty_vagrant_no_consumer_goods days = -1 } 
		}
		if = {
			limit = {
				owner = {	
					AND = {
					is_country_type = default_ev
					has_country_resource = { type = alloys amount < 1 }
					has_monthly_income = { resource = alloys value < 0 }
					}
				}
			}
			add_modifier = { modifier = empty_vagrant_no_alloys days = -1 } 
		}
		if = {
			limit = {
				owner = {	
					AND = {
					is_country_type = default_ev
					has_country_resource = { type = energy amount < 1 }
					has_monthly_income = { resource = energy value < 0 }
					}
				}
			}
			add_modifier = { modifier = empty_vagrant_no_energy days = -1 } 
		}
		if = {
			limit = {
				AND = {
					is_country_type = default_ev
					has_country_resource = { type = volatile_motes amount <= 0 }
					has_monthly_income = { resource = volatile_motes value < 0 }
				}
			}
			add_modifier = { modifier = empty_vagrant_no_volatile_motes days = -1 } 
		}
		if = {
			limit = {
				AND = {
					is_country_type = default_ev
					has_country_resource = { type = exotic_gases amount <= 0 }
					has_monthly_income = { resource = exotic_gases value < 0 }
				}
			}
			add_modifier = { modifier = empty_vagrant_no_exotic_gases days = -1 } 
		}
		if = {
			limit = {
				AND = {
					is_country_type = default_ev
					has_country_resource = { type = rare_crystals amount <= 0 }
					has_monthly_income = { resource = rare_crystals value < 0 }
				}
			}
			add_modifier = { modifier = empty_vagrant_no_rare_crystals days = -1 } 
		}
		
		if = {
			limit = {
				owner = {	
					OR = {
						AND = {
							is_country_type = default_ev
							has_country_resource = { type = sr_living_metal amount <= 0 }
							has_monthly_income = { resource = sr_living_metal value < 0 }
						}
						AND = {
							is_country_type = default_ev
							has_country_resource = { type = sr_zro amount <= 0 }
							has_monthly_income = { resource = sr_zro value < 0 }
						}
						AND = {
							is_country_type = default_ev
							has_country_resource = { type = sr_dark_matter amount <= 0 }
							has_monthly_income = { resource = sr_dark_matter value < 0 }
						}
						AND = {
							is_country_type = default_ev
							has_country_resource = { type = nanites amount <= 0 }
							has_monthly_income = { resource = nanites value < 0 }
						}
					}
				}
			}						
			add_modifier = { modifier = empty_vagrant_no_advanced_resources days = -1 } 
		}
	}
}
